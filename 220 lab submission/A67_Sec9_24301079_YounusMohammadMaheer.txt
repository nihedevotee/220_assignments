    public static void swapChild( BTNode root, int lvl, int M ){
        if (root==null){return;}
        if (lvl < M) {
            BTNode temp= root.left;
            root.left=root.right;
            root.right=temp;
        }
            swapChild(root.left, lvl+1, M);
            swapChild(root.right, lvl+1, M);
        

        
    }

    public static Integer subtractSummation( BTNode root ){
        if (root == null) {
            return 0; 
        }
        int left=sumLeft(root.left);
        int right=sumRight(root.right);


        return left-right; 
    }

    public static int sumLeft(BTNode node) {
        if (node == null) {
            return 0; 
        }
        
        return (Integer) node.elem + sumLeft(node.left) + sumLeft(node.right);
    }
    
    public static int sumRight(BTNode node) {
        if (node == null) {
            return 0; 
        }
        
        return (Integer) node.elem + sumRight(node.left) + sumRight(node.right);
    }

}

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        int sum=0;
        if (root==null )return 0;
        
        if(root.elem>=low && root.elem<=high){
            sum=sum+ root.elem;
        }
        
        return sum+rangeSum(root.left, low, high)+rangeSum(root.right, low, high);
    }

    public static Integer mirrorSum( BSTNode root ){
        //int sum=0;
        //if (root==null)return 0;

        BSTNode l=root.left;
        BSTNode r=root.right;
        
        return traverse (l,r);
    }

    public static Integer traverse( BSTNode rootL, BSTNode rootR ){
        if (rootL==null || rootR==null)return 0;
        int sum=0;
        
        sum = rootL.elem + rootR.elem;
        return sum+traverse (rootL.left,rootR.right)+traverse (rootL.right,rootR.left);

    }
}
